chrome storage class:

class storageData{
    //type being session, sync, or local
    constructor(key, type){
        this.key = key;
        this.type = type;
    }
    
    get(){
        chrome.storage[this.type].get([this.key]);
    }
    set(value){
        chrome.storage[this.type].set({[this.key]: value});
    }
}
